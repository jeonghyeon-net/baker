# Baker 구현 계획

> **에이전트 워커용:** REQUIRED SUB-SKILL: 이 계획은 `- [ ]` 체크박스 문법으로 진행 상황을 추적하므로, 작업별로 구현할 때는 subagent-driven-development(권장) 또는 executing-plans를 사용한다.

**목표:** `baker`라는 Go 기반 TUI를 만들어 Git workspace/worktree 생성·삭제·선택과 shell `cd` 연동을 안전하게 처리한다.

**아키텍처:** 루트 엔트리포인트는 `cmd/baker/main.go` 하나로 유지하고, 실제 로직은 `internal/config`, `internal/shell`, `internal/git`, `internal/github`, `internal/workspace`, `internal/worktree`, `internal/ui`, `internal/app`로 분리한다. Git/GitHub 동작은 각각 `git`/`gh` CLI를 얇게 감싼 서비스로 구현하고, shell `cd`는 rc 파일에 설치되는 함수 + 결과 파일 프로토콜로 처리한다.

**기술 스택:** Go 1.26.0, Bubble Tea, Bubbles, Lip Gloss, 표준 라이브러리 `testing`, `git` CLI, `gh` CLI

```go
package config

import (
    "encoding/json"
    "errors"
    "os"
    "path/filepath"

    "github.com/jeonghyeon-net/baker/internal/domain"
)

type Registry struct {
    Workspaces []domain.Workspace `json:"workspaces"`
}

func LoadRegistry(path string) (Registry, error) {
    data, err := os.ReadFile(path)
    if errors.Is(err, os.ErrNotExist) {
        return Registry{}, nil
    }
    if err != nil {
        return Registry{}, err
    }

    var registry Registry
    if err := json.Unmarshal(data, &registry); err != nil {
        return Registry{}, err
    }
    return registry, nil
}

func SaveRegistry(path string, registry Registry) error {
    if err := os.MkdirAll(filepath.Dir(path), 0o755); err != nil {
        return err
    }

    data, err := json.MarshalIndent(registry, "", "  ")
    if err != nil {
        return err
    }

    tempFile := path + ".tmp"
    if err := os.WriteFile(tempFile, data, 0o644); err != nil {
        return err
    }
    return os.Rename(tempFile, path)
}
```

```go
package shell

import (
    "fmt"
    "path/filepath"
    "strings"
)

func Detect(shellPath string, home string) (string, string, error) {
    switch filepath.Base(shellPath) {
    case "zsh":
        return "zsh", filepath.Join(home, ".zshrc"), nil
    case "bash":
        return "bash", filepath.Join(home, ".bashrc"), nil
    default:
        return "", "", fmt.Errorf("unsupported shell: %s", shellPath)
    }
}

func renderHook(shellName string) string {
    functionBody := `# >>> baker initialize >>>
baker() {
  local baker_result_file
  baker_result_file="$(mktemp)"
  command baker __shell --result-file "$baker_result_file" "$@"
  local baker_status=$?
  if [ $baker_status -eq 0 ] && [ -s "$baker_result_file" ]; then
    local baker_target
    baker_target="$(cat "$baker_result_file")"
    cd "$baker_target" || return $?
  fi
  rm -f "$baker_result_file"
  return $baker_status
}
# <<< baker initialize <<<
`
    return strings.ReplaceAll(functionBody, "\r\n", "\n")
}
```

```go
package shell

import (
    "os"
    "strings"
)

func InstallHook(rcPath string, shellName string) error {
    hook := renderHook(shellName)

    data, err := os.ReadFile(rcPath)
    if err != nil && !os.IsNotExist(err) {
        return err
    }
    if strings.Contains(string(data), "# >>> baker initialize >>>") {
        return nil
    }

    content := string(data)
    if content != "" && !strings.HasSuffix(content, "\n") {
        content += "\n"
    }
    content += hook
    return os.WriteFile(rcPath, []byte(content), 0o644)
}
```

```go
package exec

import (
    "bytes"
    "context"
    osExec "os/exec"
)

type Result struct {
    Stdout string
    Stderr string
}

type Runner interface {
    Run(ctx context.Context, name string, args ...string) (Result, error)
}

type CommandRunner struct{}

func (CommandRunner) Run(ctx context.Context, name string, args ...string) (Result, error) {
    cmd := osExec.CommandContext(ctx, name, args...)
    var stdout bytes.Buffer
    var stderr bytes.Buffer
    cmd.Stdout = &stdout
    cmd.Stderr = &stderr
    err := cmd.Run()
    return Result{Stdout: stdout.String(), Stderr: stderr.String()}, err
}
```

```go
package git

import (
    "fmt"
    "strings"

    "github.com/jeonghyeon-net/baker/internal/domain"
)

func ParseBranches(output string) ([]domain.BranchRef, error) {
    var branches []domain.BranchRef
    for _, line := range strings.Split(strings.TrimSpace(output), "\n") {
        if strings.TrimSpace(line) == "" {
            continue
        }
        parts := strings.Split(line, "\t")
        if len(parts) != 3 {
            return nil, fmt.Errorf("invalid branch line: %q", line)
        }
        branches = append(branches, domain.BranchRef{
            Name:       parts[0],
            Source:     parts[1],
            RemoteName: parts[2],
        })
    }
    return branches, nil
}

func ParseWorktrees(output string) ([]domain.Worktree, error) {
    blocks := strings.Split(strings.TrimSpace(output), "\n\n")
    worktrees := make([]domain.Worktree, 0, len(blocks))

    for _, block := range blocks {
        if strings.TrimSpace(block) == "" {
            continue
        }
        wt := domain.Worktree{}
        for _, line := range strings.Split(block, "\n") {
            switch {
            case strings.HasPrefix(line, "worktree "):
                wt.Path = strings.TrimPrefix(line, "worktree ")
            case strings.HasPrefix(line, "HEAD "):
```

```go
    if err != nil {
        return nil, err
    }

        return nil, err

    remoteOutput := strings.TrimSpace(remoteResult.Stdout)
```

```go
func (s Service) CreateFromRemoteURL(ctx context.Context, remoteURL string, workspaceName string) (domain.Workspace, error) {
    owner, repo, err := parseRemote(remoteURL)
    if err != nil {
        return domain.Workspace{}, err
    }

    repositoryPath := filepath.Join(s.Paths.RepositoriesRoot, workspaceName)
    if err := s.Git.CloneBare(ctx, remoteURL, repositoryPath); err != nil {
        return domain.Workspace{}, err
    }

    return domain.Workspace{
        Name:           workspaceName,
        RemoteURL:      remoteURL,
        Owner:          owner,
        Repo:           repo,
        RepositoryPath: repositoryPath,
    }, nil
}

func (s Service) CreateFromGitHubRepo(ctx context.Context, repo domain.GitHubRepo, workspaceName string) (domain.Workspace, error) {
    workspace, err := s.CreateFromRemoteURL(ctx, repo.SSHURL, workspaceName)
    if err != nil {
        return domain.Workspace{}, err
    }
    workspace.DefaultBranch = repo.DefaultBranch
    return workspace, nil
}

func (s Service) Sync(ctx context.Context, workspace domain.Workspace) error {
    return s.Git.FetchAll(ctx, workspace.RepositoryPath)
}

func parseRemote(remoteURL string) (string, string, error) {
    trimmed := strings.TrimSuffix(strings.TrimPrefix(remoteURL, "git@github.com:"), ".git")
    parts := strings.Split(trimmed, "/")
    if len(parts) != 2 {
        return "", "", fmt.Errorf("invalid GitHub remote: %s", remoteURL)
    }
    return parts[0], parts[1], nil
```

```go
type DeleteMode string

const (
    DeleteModeWorktreeOnly DeleteMode = "worktree-only"
    DeleteModeLocalBranch  DeleteMode = "worktree-and-local-branch"
    DeleteModeAll          DeleteMode = "worktree-local-and-remote-branch"
)

type GitClient interface {
    AddExistingBranchWorktree(ctx context.Context, repoPath string, branch string, worktreePath string) error
    AddNewBranchWorktree(ctx context.Context, repoPath string, baseBranch string, newBranch string, worktreePath string) error
    PushBranch(ctx context.Context, worktreePath string, branch string) error
    RemoveWorktree(ctx context.Context, repoPath string, worktreePath string, force bool) error
    DeleteLocalBranch(ctx context.Context, repoPath string, branch string, force bool) error
    DeleteRemoteBranch(ctx context.Context, worktreePath string, branch string) error
}

type CreateResult struct {
    Path    string
    Partial bool
}

type Service struct {
    Git   GitClient
    Paths config.Paths
}

func (s Service) CreateFromExistingBranch(ctx context.Context, workspace domain.Workspace, branches []domain.BranchRef, branch string, worktreeName string) (CreateResult, error) {
    for _, ref := range branches {
        if ref.Name == branch && ref.HasActiveWorktree {
            return CreateResult{}, fmt.Errorf("branch %s already has an active worktree", branch)
        }
    }
    path := filepath.Join(s.Paths.WorktreesRoot, workspace.Name, worktreeName)
    if err := s.Git.AddExistingBranchWorktree(ctx, workspace.RepositoryPath, branch, path); err != nil {
        return CreateResult{}, err
    }
    return CreateResult{Path: path}, nil
}

func (s Service) CreateFromNewBranch(ctx context.Context, workspace domain.Workspace, baseBranch string, newBranch string, worktreeName string) (CreateResult, error) {
    path := filepath.Join(s.Paths.WorktreesRoot, workspace.Name, worktreeName)
    if err := s.Git.AddNewBranchWorktree(ctx, workspace.RepositoryPath, baseBranch, newBranch, path); err != nil {
        return CreateResult{}, err
    }
    if err := s.Git.PushBranch(ctx, path, newBranch); err != nil {
        return CreateResult{Path: path, Partial: true}, err
    }
    return CreateResult{Path: path}, nil
}

func (s Service) Delete(ctx context.Context, workspace domain.Workspace, worktree domain.Worktree, mode DeleteMode, force bool) error {
    if err := s.Git.RemoveWorktree(ctx, workspace.RepositoryPath, worktree.Path, force); err != nil {
        return err
    }
    switch mode {
    case DeleteModeWorktreeOnly:
        return nil
    case DeleteModeLocalBranch:
        return s.Git.DeleteLocalBranch(ctx, workspace.RepositoryPath, worktree.BranchName, force)
    case DeleteModeAll:
        if err := s.Git.DeleteLocalBranch(ctx, workspace.RepositoryPath, worktree.BranchName, force); err != nil {
            return err
        }
        return s.Git.DeleteRemoteBranch(ctx, worktree.Path, worktree.BranchName)
    default:
        return fmt.Errorf("unknown delete mode: %s", mode)
    }
}
```

```go
package ui

import tea "github.com/charmbracelet/bubbletea"

type Screen string

const (
    ScreenWorktrees Screen = "worktrees"
)

type State struct {
    Screen       Screen
    Worktrees    []string
    Cursor       int
    SelectedPath string
}

type Model struct {
    State
}

func NewModel(state State) Model {
    return Model{State: state}
}

func (m Model) Init() tea.Cmd { return nil }

func (m Model) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
    switch msg := msg.(type) {
    case tea.KeyMsg:
        switch msg.Type {
        case tea.KeyUp:
            if m.Cursor > 0 {
                m.Cursor--
            }
        case tea.KeyDown:
            if m.Cursor < len(m.Worktrees)-1 {
                m.Cursor++
            }
        case tea.KeyEnter:
            if len(m.Worktrees) > 0 {
                m.SelectedPath = m.Worktrees[m.Cursor]
            }
        }
    }
    return m, nil
}

func (m Model) View() string {
    if len(m.Worktrees) == 0 {
        return "No worktrees"
    }
    return m.Worktrees[m.Cursor]
}
```

```go
const (
    ScreenWorktrees           Screen = "worktrees"
    ScreenCreateWorktree      Screen = "create-worktree"
    ScreenWorkspaceGitHubPicker Screen = "workspace-github-picker"
    ScreenDeleteConfirm       Screen = "delete-confirm"
)

type State struct {
    Screen       Screen
    Worktrees    []string
    Actions      []string
    DeleteModes  []string
    Cursor       int
    SelectedPath string
}

func (m Model) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
    switch msg := msg.(type) {
    case tea.KeyMsg:
        switch msg.Type {
        case tea.KeyUp:
            if m.Cursor > 0 {
                m.Cursor--
            }
        case tea.KeyDown:
            limit := len(m.Worktrees)
            if len(m.Actions) > 0 {
                limit = len(m.Actions)
            }
            if m.Cursor < limit-1 {
                m.Cursor++
            }
        case tea.KeyEnter:
            if len(m.Actions) > 0 {
                switch m.Actions[m.Cursor] {
                case "create-worktree":
                    m.Screen = ScreenCreateWorktree
                case "create-workspace-github":
                    m.Screen = ScreenWorkspaceGitHubPicker
                case "delete-worktree":
                    m.Screen = ScreenDeleteConfirm
                    m.DeleteModes = []string{"worktree-only", "worktree-and-local-branch", "worktree-local-and-remote-branch"}
                case "open":
                    if len(m.Worktrees) > 0 {
                        m.SelectedPath = m.Worktrees[0]
                    }
                }
                return m, nil
            }
            if len(m.Worktrees) > 0 {
                m.SelectedPath = m.Worktrees[m.Cursor]
            }
        }
    }
    return m, nil
}
```

```go
package app

import (
    "context"
    "os"
)

type Mode string

const (
    ModeNeedsSource Mode = "needs-source"
    ModeInteractive Mode = "interactive"
)

type Result struct {
    Mode    Mode
    Message string
}

type Options struct {
    ResultFile string
}

type ShellInstaller interface {
    Ensure() (ready bool, message string, err error)
}

type Application struct {
    Shell ShellInstaller
}

func (a Application) Run(ctx context.Context, opts Options) (Result, error) {
    _ = ctx
    ready, message, err := a.Shell.Ensure()
    if err != nil {
        return Result{}, err
    }
    if !ready {
        return Result{Mode: ModeNeedsSource, Message: message}, nil
    }
    return Result{Mode: ModeInteractive}, nil
}

func WriteShellResult(path string, target string) error {
    return os.WriteFile(path, []byte(target), 0o644)
}
```

```go
package main

import (
    "context"
    "flag"
    "fmt"
    "os"

    "github.com/jeonghyeon-net/baker/internal/app"
)

type bootstrapShell struct{}

func (bootstrapShell) Ensure() (bool, string, error) {
    return false, "source ~/.zshrc", nil
}

func main() {
    if len(os.Args) > 1 && os.Args[1] == "__shell" {
        shellFlags := flag.NewFlagSet("__shell", flag.ExitOnError)
        resultFile := shellFlags.String("result-file", "", "path to shell result file")
        _ = shellFlags.Parse(os.Args[2:])

        if *resultFile != "" {
            if target := os.Getenv("BAKER_SELECTED_PATH"); target != "" {
                if err := app.WriteShellResult(*resultFile, target); err != nil {
                    fmt.Fprintln(os.Stderr, err)
                    os.Exit(1)
                }
            }
        }
        return
    }

    result, err := app.Application{Shell: bootstrapShell{}}.Run(context.Background(), app.Options{})
    if err != nil {
        fmt.Fprintln(os.Stderr, err)
        os.Exit(1)
    }
    if result.Message != "" {
        fmt.Println(result.Message)
    }
}
```
